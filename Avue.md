# avue-form

## 数据字典的使用
树下拉选择框
```
{
          label: "归属机构",
          prop: "deptId",
          type: "tree",
          rules: [{
            required: true,
            message: "归属机构不能为空",
            trigger: "blur"
          }],
          dicUrl: "/admin/dept/tree",
          props: {
            label: "deptName",
            value: "id",
            children: "children"
          }
        },
        
        
{
    label: "证件类型",
    prop: "certificateType",
    type: "select",
    filterable: true,//搜索
    dicUrl: "/admin/dict/type/sys_certificate_type"
},
```
